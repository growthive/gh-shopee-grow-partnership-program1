        // Memastikan elemen fungsi tersedia secara global (mengatasi error ReferenceError di preview)

        window.openFormModal = function(paket) {

            // Trigger tracking Initiate Checkout Meta Pixel

            if (typeof fbq === 'function') {

                fbq('track', 'InitiateCheckout');

            }


            const modal = document.getElementById('formModal');

            const packageText = document.getElementById('selectedPackageText');

            

            if(paket !== 'General') {

                packageText.innerText = `Paket Diminati: ${paket}`;

                packageText.classList.remove('hidden');

            } else {

                packageText.classList.add('hidden');

            }

            

            modal.classList.remove('hidden');

            modal.classList.add('flex');

            

            // Memberikan sedikit jeda untuk animasi Tailwind

            setTimeout(() => {

                modal.classList.remove('opacity-0');

                modal.querySelector('div').classList.remove('scale-95');

                modal.querySelector('div').classList.add('scale-100');

            }, 10);

            

            // Mencegah halaman utama bisa di-scroll saat modal terbuka

            document.body.style.overflow = 'hidden';

        };


        window.closeFormModal = function() {

            const modal = document.getElementById('formModal');

            

            // Memulai animasi penutupan

            modal.classList.add('opacity-0');

            modal.querySelector('div').classList.remove('scale-100');

            modal.querySelector('div').classList.add('scale-95');

            

            // Menyembunyikan elemen setelah animasi selesai (300ms)

            setTimeout(() => {

                modal.classList.add('hidden');

                modal.classList.remove('flex');

            }, 300);

            

            // Mengembalikan fungsi scroll pada halaman utama

            document.body.style.overflow = 'auto';

        };


        // Menutup modal jika user melakukan klik di luar area putih form

        window.onclick = function(event) {

            const modal = document.getElementById('formModal');

            if (event.target === modal) {

                window.closeFormModal();

            }

        };
